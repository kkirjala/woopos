49      /**
50       * Fire up the application.
51       * @param backend the backend database to be used for storing information.
52       * @param ui User interface class
53       * @param paymentMethods the allowed/used payment methods.
54       */
55      public WooPOS(PosBackend backend, PosUI ui, List<PaymentMethod> paymentMethods) {
56  
57          this.backend = backend;
58          this.ui = ui;
59          this.paymentMethods = paymentMethods;
60          
61          
62          this.orderController = new OrderController(this);        
63  
64          fireOnPosStartup();
65  
66          // TODO: wait for the UI to close
67          fireOnPosClose();
68  
69      }
70  
71      /**
72       * Firing of the application start event.
73       */
74      public void fireOnPosStartup() {
75  
76          this.backend.onPosStartup(this);
77  
78          for (PaymentMethod currPaymentMethod : this.paymentMethods) {
79              currPaymentMethod.onPosStartup(this);
80          }
81          
82          this.ui.onPosStartup(this);
83          
84          this.orderController.onPosStartup(this);
85  
86      }
87  
88      /**
89       * Fire the application close event.
90       */
91      public void fireOnPosClose() {
92  
93          this.backend.onPosClose(this);
94  
95          for (PaymentMethod currPaymentMethod : this.paymentMethods) {
96              currPaymentMethod.onPosClose(this);
97          }
98          
99          this.ui.onPosClose(this);
100         
101         this.orderController.onPosClose(this);
102 
103     }
