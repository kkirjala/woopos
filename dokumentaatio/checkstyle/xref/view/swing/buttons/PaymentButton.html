1   package view.swing.buttons;

3   import interfaces.PaymentMethod;
4   import javax.swing.JButton;
5   
6   /**
7    * A custom button class for creating buttons that trigger making Payments
8    * for Orders.
9    * @author kkirjala
10   */
11  public class PaymentButton extends JButton {
12  
13      private PaymentMethod paymentMethod;
14  
15      /**
16       * Create a new button.
17       * @param paymentMethod the PaymentMethod that will be triggered by this
18       * button.
19       */
20      public PaymentButton(PaymentMethod paymentMethod) {
21  
22          super(paymentMethod.getDisplayName());
23  
24          this.paymentMethod = paymentMethod;
25          this.setActionCommand("create_payment");
26  
27      }
28  
29      /**
30       * The PaymentMethod associated with this button.
31       * @return the PaymentMethod.
32       */
33      public PaymentMethod getPaymentMethod() {
34          return paymentMethod;
35      }
36  
37  }
